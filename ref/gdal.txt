# To import a gpx file into a mysql dbase...

DBASE=geo
USER=rsigler
PASSWD=

OGR2OGR=/usr/bin/ogr2ogr

if [ ! -f $1 ]; then
	echo "expected a gpx file as param 1"
	exit
fi

SUM=`md4sum $1 | awk '{print $1}'`

$OGR2OGR \
-f "MySQL" \
MYSQL:"$DBASE,user=$USER,passwd=$PASSWD" \
-lco engine=MYISAM \
-nln TRACK_$SUM \
-overwrite \
$1
